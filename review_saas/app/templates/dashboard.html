# filename: app/routes/dashboard.py # ========================================================== # REVIEW INTELLIGENCE DASHBOARD — ENTERPRISE / FRONTEND‑SYNCED # Fully compatible with provided Dashboard.html # ========================================================== from __future__ import annotations import io import os import asyncio import logging from datetime import datetime from collections import Counter, defaultdict from typing import Dict, List, Any from fastapi import APIRouter, Depends, Query, HTTPException from fastapi.responses import JSONResponse, StreamingResponse from sqlalchemy.ext.asyncio import AsyncSession from sqlalchemy import select, desc from sqlalchemy.orm import load_only from reportlab.lib.pagesizes import A4 from reportlab.pdfgen import canvas from reportlab.pdfbase import pdfmetrics from reportlab.pdfbase.cidfonts import UnicodeCIDFont from openai import OpenAI from app.core.db import get_db from app.core.models import Company, Review # ---------------------------------------------------------- # CONFIGURATION # ---------------------------------------------------------- router = APIRouter(tags=["Dashboard"]) logger = logging.getLogger("dashboard") client = OpenAI(api_key=os.getenv("OPENAI_API_KEY")) NEGATIVE_RATINGS = {1, 2} NEG_SENTIMENT_LIMIT = -0.2 MIN_SAMPLE = 20 STOPWORDS = { "the","and","with","this","that","for","from","were", "have","has","very","just","they","them","their", "there","but","not","your","will","can","our" } pdfmetrics.registerFont(UnicodeCIDFont("HYSMyeongJo-Medium")) # ---------------------------------------------------------- # HELPERS # ---------------------------------------------------------- def safe_date(v: str | None) -> datetime | None: try: return datetime.fromisoformat(v) if v else None except Exception: return None def clean_keywords(text: str) -> List[str]: if not text: return [] out = [] for w in text.lower().split(): w = w.strip(".,!?()[]{}'\"") if len(w) >= 4 and w.isalpha() and w not in STOPWORDS: out.append(w) return out async def get_company(db: AsyncSession, cid: int) -> Company: res = await db.execute(select(Company).where(Company.id == cid)) company = res.scalars().first() if not company: raise HTTPException(404, "Company not found") return company # ---------------------------------------------------------- # CORE ANALYTICS (MATCHES UI KEYS EXACTLY) # ---------------------------------------------------------- def compute_analytics(reviews: List[Review]) -> Dict[str, Any]: if not reviews: return { "metadata": {"total_reviews": 0}, "kpis": { "average_rating": 0, "churn_prediction": 0, "loyalty_score": 0, "reputation_score": 100 }, "visualizations": { "emotions": {"Positive": 0, "Neutral": 0, "Negative": 0}, "ratings": {i: 0 for i in range(1, 6)}, "sentiment_trend": [] }, "negative_reviews": [], "top_keywords": [] } ratings = [] rating_dist = {i: 0 for i in range(1, 6)} emo = {"Positive": 0, "Neutral": 0, "Negative": 0} monthly = defaultdict(list) negative_reviews = [] neg_count = severe_count = 0 for r in reviews: if r.rating: rating_dist[int(r.rating)] += 1 ratings.append(int(r.rating)) if r.google_review_time: monthly[r.google_review_time.strftime("%b %Y")].append(int(r.rating)) if int(r.rating) in NEGATIVE_RATINGS: severe_count += 1 if r.text and len(negative_reviews) < 6: negative_reviews.append({ "author": r.author_name or "Anonymous", "rating": r.rating, "text": r.text[:160] + ("..." if len(r.text) > 160 else ""), "date": r.google_review_time.strftime("%Y-%m-%d") if r.google_review_time else "N/A" }) if r.sentiment_score is not None: if r.sentiment_score <= NEG_SENTIMENT_LIMIT: emo["Negative"] += 1 neg_count += 1 elif r.sentiment_score >= abs(NEG_SENTIMENT_LIMIT): emo["Positive"] += 1 else: emo["Neutral"] += 1 total = len(reviews) eff_total = max(total, MIN_SAMPLE) risk_pct = round(((neg_count * 0.6 + severe_count * 0.4) / eff_total) * 100, 1) trend = [ {"month": k, "avg": round(sum(v) / len(v), 2), "count": len(v)} for k, v in monthly.items() ] return { "metadata": {"total_reviews": total}, "kpis": { "average_rating": round(sum(ratings) / len(ratings), 2), "churn_prediction": round(min(95, risk_pct), 1), "loyalty_score": round((emo["Positive"] / total) * 100, 1), "reputation_score": max(0, round(100 - risk_pct, 1)) }, "visualizations": { "emotions": emo, "ratings": rating_dist, "sentiment_trend": sorted(trend, key=lambda x: x["month"]) }, "negative_reviews": negative_reviews } # ========================================================== # ROUTES (MATCH FRONTEND API CONSTANTS) # ========================================================== @router.get("/api/insights", response_class=JSONResponse) async def insights( company_id: int = Query(...), start: str | None = Query(None), end: str | None = Query(None), db: AsyncSession = Depends(get_db) ): await get_company(db, company_id) stmt = select(Review).options( load_only( Review.rating, Review.text, Review.sentiment_score, Review.google_review_time, Review.author_name ) ).where(Review.company_id == company_id) if sd := safe_date(start): stmt = stmt.where(Review.google_review_time >= sd) if ed := safe_date(end): stmt = stmt.where(Review.google_review_time <= ed) res = await db.execute(stmt) reviews = res.scalars().all() analytics = compute_analytics(reviews) keywords = [] for r in reviews: w = 2 if r.sentiment_score is not None and r.sentiment_score <= NEG_SENTIMENT_LIMIT else 1 keywords.extend(clean_keywords(r.text) * w) analytics["top_keywords"] = [k for k, _ in Counter(keywords).most_common(25)] return analytics @router.get("/api/revenue", response_class=JSONResponse) async def revenue(company_id: int, db: AsyncSession = Depends(get_db)): await get_company(db, company_id) res = await db.execute(select(Review).where(Review.company_id == company_id)) analytics = compute_analytics(res.scalars().all()) risk_pct = analytics["kpis"]["churn_prediction"] return { "risk_percent": risk_pct, "impact": "High" if risk_pct > 25 else "Medium" if risk_pct > 12 else "Low", "reputation_score": analytics["kpis"]["reputation_score"] } @router.post("/api/chatbot/explain", response_class=JSONResponse) async def chatbot( payload: dict, company_id: int = Query(...), db: AsyncSession = Depends(get_db) ): msg = (payload.get("message") or "").strip() if not msg: return {"answer": "Ask a clear business or customer‑experience question."} res = await db.execute(select(Review).where(Review.company_id == company_id).limit(200)) analytics = compute_analytics(res.scalars().all()) prompt = ( f"Act as a senior CX strategy consultant.\n" f"Reviews: {analytics['metadata']['total_reviews']}\n" f"Average rating: {analytics['kpis']['average_rating']}\n" f"Risk level: {analytics['kpis']['churn_prediction']}%\n\n" f"Question: {msg}" ) try: response = await asyncio.wait_for( asyncio.to_thread( lambda: client.chat.completions.create( model="gpt-4o-mini", messages=[{"role": "user", "content": prompt}], temperature=0.25, max_tokens=300 ) ), timeout=15 ) return {"answer": response.choices[0].message.content.strip()} except Exception as e: logger.error(e) return {"answer": "AI analysis temporarily unavailable."} @router.get("/api/recent-reviews/{company_id}", response_class=JSONResponse) async def recent_reviews(company_id: int, db: AsyncSession = Depends(get_db)): await get_company(db, company_id) res = await db.execute( select(Review) .where(Review.company_id == company_id) .order_by(desc(Review.google_review_time)) .limit(100) ) return [ { "author": r.author_name or "Anonymous", "rating": r.rating, "text": r.text, "date": r.google_review_time.strftime("%Y-%m-%d") if r.google_review_time else "N/A" } for r in res.scalars().all() ] @router.get("/api/executive-report/pdf/{company_id}") async def media_type="application/pdf")
