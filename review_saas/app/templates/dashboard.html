────────────────────────────────────────────────────────────────────────────── # NLP & Sentiment Logic (Negation Aware) # ────────────────────────────────────────────────────────────────────────────── _RATING_PROXY = {5: 0.8, 4: 0.4, 3: 0.0, 2: -0.4, 1: -0.8} _NEGATORS = {"not", "never", "no", "hardly", "barely", "scarcely", "without", "lack"} _TOKEN_RE = re.compile(r"[a-zA-Z]+") _LEXICON = { "great": 4, "excellent": 5, "good": 3, "friendly": 3, "clean": 2, "amazing": 4, "bad": -3, "poor": -3, "worst": -5, "slow": -2, "dirty": -3, "rude": -4, "issue": -2 } def _tokenize(text: str) -> List[str]: if not text: return [] toks = [t.lower() for t in _TOKEN_RE.findall(text)] return [t for t in toks if len(t) > 2] def _safe_sentiment(text: str, rating: Optional[int] = None) -> float: tokens = _tokenize(text or "") score = 0.0 for i, t in enumerate(tokens): base = _LEXICON.get(t, 0) if base > 0 and any(tokens[j] in _NEGATORS for j in range(max(0, i-3), i)): base = -abs(base) score += base norm = score / max(1.0, math.sqrt(len(tokens) or 1)) lex = float(max(-1.0, min(1.0, norm / 5.0))) if rating is None: return lex proxy = _RATING_PROXY.get(int(rating), 0.0) return float(max(-1.0, min(1.0, 0.65 * lex + 0.35 * proxy))) #
